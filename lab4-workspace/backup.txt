void SPI_SendByte(unsigned char byte) {

  unsigned char counter;

  for (counter = 0; counter < 8; counter++) {
    SPI_SCK_0;

    if ((byte & 0x80) == 0) {
      SPI_SDA_0;
    } else
      SPI_SDA_1;
    byte = byte << 1;

    SPI_SCK_1;

    SPI_SCK_0;
  }
}
